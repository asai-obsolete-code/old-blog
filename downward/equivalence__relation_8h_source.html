129 template<class T, class Equal>
130 EquivalenceRelation *EquivalenceRelation::from_annotated_elements(int n,
131  std::vector<std::pair<T, int> > &annotated_elements) {
132  EquivalenceRelation *relation = new EquivalenceRelation(n);
133  if (!annotated_elements.empty()) {
134  sort(annotated_elements.begin(), annotated_elements.end());
135  Equal equal;
136  T current_class_label = annotated_elements[0].first;
137  BlockListIter it_current_block = relation->add_empty_block();
138  for (size_t i = 0; i < annotated_elements.size(); ++i) {
139  T label = annotated_elements[i].first;
140  int element = annotated_elements[i].second;
141  if (!equal(label, current_class_label)) {
142  current_class_label = label;
143  it_current_block = relation->add_empty_block();
144  }
145  ElementListIter it_element = it_current_block->insert(element);
146  relation->element_positions[element] = make_pair(it_current_block, it_element);
147  }
148  }
149  return relation;
150 }
