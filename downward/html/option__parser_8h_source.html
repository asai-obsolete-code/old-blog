4 /*
5 How to add support for a new type NT to the option parser:
6 In this file, tell the parser how to parse NT:
7 - Create a template specialization TokenParser<NT> (or overload
8  the ">>"-operator for NT).
9  See the existing specializations for examples.
10 
11 If you want classes of type NT to be included in the help output:
12 In option_parser_util.h:
13 - Create a template specialization DefaultValueNamer<NT>, or overload
14  the "<<"-operator for NT. Not necessary if your type will never have
15  a default value.
16 - (optional) Create a template specialization for TypeNamer<NT> - you
17  don't have to do this, but it might print the typename in a
18  more readable way.
19 In option_parser.cc:
20 - add NT to the functions static void get_help(string k) and
21  static void get_full_help()
22 
23 If NT shall be predefinable:
24 - See the functions predefine_lmgraph() and predefine_heuristic()
25  in option parser.cc for examples.
26  You will also need to extend OptionParser::parse_cmd_line(), and
27  should add an explanation to OptionParser::usage().
28 */

249 template <class T>
250 void OptionParser::add_option(
251  std::string k,
252  std::string h, std::string default_value) {
253  if (help_mode()) {
254  DocStore::instance()->add_arg(parse_tree.begin()->value,
255  k, h,
256  TypeNamer<T>::name(), default_value);
257  return;
258  }
259  valid_keys.push_back(k);
260  bool use_default(false);
261  ParseTree::sibling_iterator arg = next_unparsed_argument;
262  //scenario where we have already handled all arguments
263  if (arg == parse_tree.end(parse_tree.begin())) {
264  if (default_value.empty()) {
265  error("missing option: " + k);
266  } else if (default_value == NONE) {
267  return;
268  } else {
269  use_default = true;
270  }
271  } else {
272  //handling arguments with explicit keyword:
273  if (!arg->key.empty()) { //to check if we reached the params supplied with key
274  //try to find a parameter passed with keyword k
275  for (; arg != parse_tree.end(parse_tree.begin()); ++arg) {
276  if (arg->key.compare(k) == 0)
277  break;
278  }
279  if (arg == parse_tree.end(parse_tree.begin())) {
280  if (default_value.empty()) {
281  error("missing option: " + k);
282  } else if (default_value == NONE) {
283  return;
284  } else {
285  use_default = true;
286  }
287  }
288  }
289  }
290  std::unique_ptr<OptionParser> subparser(
291  use_default ?
292  new OptionParser(default_value, dry_run()) :
293  new OptionParser(subtree(parse_tree, arg), dry_run()));
294  T result = TokenParser<T>::parse(*subparser);
295  opts.set<T>(k, result);
296  //if we have not reached the keyword parameters yet
297  //and did not use the default value,
298  //increment the argument position pointer
299  if (!use_default && arg->key.empty()) {
300  ++next_unparsed_argument;
301  }
302 }
