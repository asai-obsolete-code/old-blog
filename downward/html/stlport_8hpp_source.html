117 // workaround does not conflict with STLports:
118 //
119 //
120 // Harold Howe says:
121 // Borland switched to STLport in BCB6. Defining BOOST_NO_STDC_NAMESPACE with
122 // BCB6 does cause problems. If we detect C++ Builder, then don't define
123 // BOOST_NO_STDC_NAMESPACE
124 //
125 #if !defined(__BORLANDC__) && !defined(__DMC__)
126 //
127 // If STLport is using it's own namespace, and the real names are in
128 // the global namespace, then we duplicate STLport's using declarations
129 // (by defining BOOST_NO_STDC_NAMESPACE), we do this because STLport doesn't
130 // necessarily import all the names we need into namespace std::
131 //
132 # if (defined(__STL_IMPORT_VENDOR_CSTD) \
133  || defined(__STL_USE_OWN_NAMESPACE) \
134  || defined(_STLP_IMPORT_VENDOR_CSTD) \
135  || defined(_STLP_USE_OWN_NAMESPACE)) \
136  && (defined(__STL_VENDOR_GLOBAL_CSTD) || defined (_STLP_VENDOR_GLOBAL_CSTD))
137 # define BOOST_NO_STDC_NAMESPACE
138 # define BOOST_NO_EXCEPTION_STD_NAMESPACE
