419 // When the standard library does not have a conforming std::use_facet there

421 // The same problem occurs with has_facet.
422 // These macros provide a consistent way to access a locale's facets.
423 // Usage:
424 // replace
425 // std::use_facet<Type>(loc);
426 // with
427 // BOOST_USE_FACET(Type, loc);
428 // Note do not add a std:: prefix to the front of BOOST_USE_FACET!
429 // Use for BOOST_HAS_FACET is analogous.

518 // they just link one instantiation of the template in the final
519 // executable). These macros provide a uniform way to cope with the
520 // problem with no effects on the calling syntax.
521 
522 // Example:
523 //
524 // #include <iostream>
525 // #include <ostream>
526 // #include <typeinfo>
527 //
528 // template <int n>
529 // void f() { std::cout << n << ' '; }

534 // int main() {
535 // f<1>();
536 // f<2>();
537 //
538 // g<int>();
539 // g<double>();
540 // }

542 // With VC++ 6.0 the output is:
543 //
544 // 2 2 double double
545 //
546 // To fix it, write
547 //
548 // template <int n>
549 // void f(BOOST_EXPLICIT_TEMPLATE_NON_TYPE(int, n)) { ... }
