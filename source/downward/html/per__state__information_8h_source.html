24 /*
25  PerStateInformation is used to associate information with states.
26  PerStateInformation<Entry> logically behaves somewhat like an unordered map
27  from states to objects of class Entry. However, lookup of unknown states is
28  supported and leads to insertion of a default value (similar to the
29  defaultdict class in Python).
30 
31  For example, landmark heuristics can use it to store the set of landmarks
32  with every state reached during search, or search algorithms can use it to
33  associate g values or creating operators with a state.
34 
35  Implementation notes: PerStateInformation is essentially implemented as a
36  kind of two-level map:
37  1. Find the correct SegmentedVector for the registry of the given state.
38  2. Look up the associated entry in the SegmentedVector based on the ID of
39  the state.
40  It is common in many use cases that we look up information for states from
41  the same registry in sequence. Therefore, to make step 1. more efficient, we
42  remember (in "cached_registry" and "cached_entries") the results of the
43  previous lookup and reuse it on consecutive lookups for the same registry.
44 
45  A PerStateInformation object subscribes to every StateRegistry for which it
46  stores information. Once a StateRegistry is destroyed, it notifies all
47  subscribed objects, which in turn destroy all information stored for states
48  in that registry.
49 */

60  /*
61  Returns the SegmentedVector associated with the given StateRegistry.
62  If no vector is associated with this registry yet, an empty one is created.
63  Both the registry and the returned vector are cached to speed up
64  consecutive calls with the same registry.
65  */
66  SegmentedVector<Entry> *get_entries(const StateRegistry *registry) {
67  if (cached_registry != registry) {
68  cached_registry = registry;
69  typename EntryVectorMap::const_iterator it = entries_by_registry.find(registry);
70  if (it == entries_by_registry.end()) {
71  cached_entries = new SegmentedVector<Entry>();
72  entries_by_registry[registry] = cached_entries;
73  registry->subscribe(this);
74  } else {
75  cached_entries = it->second;
76  }
77  }
78  assert(cached_registry == registry && cached_entries == entries_by_registry[registry]);
79  return cached_entries;
80  }

82  /*
83  Returns the SegmentedVector associated with the given StateRegistry.
84  Returns 0, if no vector is associated with this registry yet.
85  Otherwise, both the registry and the returned vector are cached to speed
86  up consecutive calls with the same registry.
87  */
88  const SegmentedVector<Entry> *get_entries(const StateRegistry *registry) const {
